      \begin{itemize}
        \item<1-> Si \(a^x=a^z \;\rightarrow\; x=z\) (funció injectiva).
        \item<2-> Ex.: si \(2^x=2^3 \;\rightarrow\; x=3\).
      \end{itemize}

      \vspace{0.15cm}
      \only<3->{
        \begin{caixadem}[width=0.92\columnwidth]
          \only<3->{$a^x=a^z$}\\[0.12cm]
          \only<4->{$a^x\cdot a^{-x}=a^z\cdot a^{-x}$}\\[0.12cm]
          \only<5->{$1=a^{z-x}$}\\[0.12cm]
          \only<6->{$z-x=0 \;\Rightarrow\; x=z$}
        \end{caixadem}
      }
    \end{column}

    % ---- EXHAUSTIVA (dreta) ----
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item<7-> \(y\in\mathbb{R}^+\), \(\exists x\in\mathbb{R}\) tal que \(a^x=y\) (funció exhaustiva en \(\mathbb{R}^+\)).
        \item<8-> Ex.: si \(f(x)=2^x\) i \(y=8 \;\rightarrow\; x=3\) perquè \(2^3=8\).
        \item<9-> Qualsevol imatge \(y\) té una única antiimatge \(x\).
        \item<10-> Ex.: per a \(y=16\) i \(f(x)=2^x\), només \(x=4\).
      \end{itemize}

      \vspace{0.08cm}
      \only<7->{
        \centering

\section{Equacions exponencials}

% ---------- DIAPO 3.0: Introducció ----------
\begin{frame}[t]
  \frametitle{3. Equacions exponencials}

  \small
  \begin{itemize}
    \item<1-> Com que \(y\in\mathbb{R}^+\), \(\exists x\in\mathbb{R}\) tal que \(a^x=y\), cada imatge \(y\) té una \textbf{única antiimatge} \(x\).
    \item<2-> Això ens permet \textbf{resoldre equacions exponencials}.
    \item<3-> Veurem \textbf{5 tipus} diferents d'equacions exponencials.
    \item<4-> Alguns exemples que resoldrem:
    \begin{itemize}
      \item<4-> \(5^x=625\)
      \item<5-> \(8^{2x+3}=4^{\frac{x}{3}}\)
      \item<6-> \(3^{x-1}+3^x+3^{x+1}=39\)
      \item<7-> \(4^x+6\cdot 2^{x+1}+32=0\)
      \item<8-> \(2^x=23\)
    \end{itemize}
  \end{itemize}
\end{frame}

% ---------- DIAPO 3.1: Tipus 1 ----------
\begin{frame}[t]
  \frametitle{3. Equacions exponencials — Tipus 1}

  \small
  \caixatitol{\(a^x=k\)}

  \vspace{0.15cm}
  \textbf{Estratègia:} expressar \(k\) com a potència de base \(a\).

  \vspace{0.2cm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \textbf{Exemple 1}
      \begin{align*}
        \only<2->{5^x &= 625} \\
        \only<3->{5^x &= 5^4} \\
        \only<4->{& \solucio{x = 4}}
      \end{align*}
    \end{column}
    \begin{column}{0.48\textwidth}
      \only<5->{\textbf{Exemple 2}}
      \begin{align*}
        \only<5->{4^x &= 64} \\
        \only<6->{4^x &= 4^3} \\
        \only<7->{& \solucio{x = 3}}
      \end{align*}
    \end{column}
  \end{columns}
\end{frame}

% ---------- DIAPO 3.2: Tipus 2 ----------
\begin{frame}[t]
  \frametitle{3. Equacions exponencials — Tipus 2}

  \small
  \caixatitol{\(a^{f(x)}=b^{g(x)}\)}

  \vspace{0.15cm}
  \textbf{Estratègia:} expressar les dues potències amb la \textbf{mateixa base}.

  \vspace{0.2cm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \textbf{Exemple 1}
      \begin{align*}
        \only<2->{3^{x+1} &= 9} \\
        \only<3->{3^{x+1} &= 3^2} \\
        \only<4->{x+1 &= 2} \\
        \only<5->{& \solucio{x = 1}}
      \end{align*}
    \end{column}
    \begin{column}{0.48\textwidth}
      \only<6->{\textbf{Exemple 2}}
      \begin{align*}
        \only<6->{8^{2x+3} &= 4^{\frac{x}{3}}} \\
        \only<7->{2^{6x+9} &= 2^{\frac{2x}{3}}} \\
        \only<8->{6x+9 &= \frac{2x}{3}} \\
        \only<9->{& \solucio{x = -\frac{27}{16}}}
      \end{align*}
    \end{column}
  \end{columns}
\end{frame}

% ---------- DIAPO 3.3: Tipus 3 ----------
\begin{frame}[t]
  \frametitle{3. Equacions exponencials — Tipus 3}

  \small
  \caixatitol{\(a^{f(x)}+a^{g(x)}+\cdots+a^{z(x)}=k\)}

  \vspace{0.15cm}
  \textbf{Estratègia:} reescriure per poder extreure \textbf{factor comú}.

  \vspace{0.2cm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \textbf{Exemple 1}
      \begin{align*}
        \only<2->{2^x+2^{x+1} &= 12} \\
        \only<3->{2^x+2^x\cdot 2^1 &= 12} \\
        \only<4->{%
          \fc{2^x}+\fc{2^x}\cdot 2 &= 12
        } \\
        \only<5->{2^x(1+2) &= 12} \\
        \only<6->{2^x &= 4} \\
        \only<7->{& \solucio{x = 2}}
      \end{align*}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[t]
  \frametitle{3. Equacions exponencials — Tipus 3 (exemple 2)}

  \small
  \textbf{Exemple 2}
  \begin{align*}
    \only<1->{3^{x-1}+3^x+3^{x+1} &= 39} \\
    \only<2->{3^x\cdot 3^{-1}+3^x+3^x\cdot 3 &= 39} \\
    \only<3->{%
      \fc{3^x}\cdot\tfrac{1}{3}+\fc{3^x}+\fc{3^x}\cdot 3 &= 39
    } \\
    \only<4->{3^x\!\left(\tfrac{1}{3}+1+3\right) &= 39} \\
    \only<5->{3^x\cdot\tfrac{13}{3} &= 39} \\
    \only<6->{3^x &= 39\cdot\tfrac{3}{13}} \\
    \only<7->{3^x &= 9} \\
    \only<8->{& \solucio{x = 2}}
  \end{align*}
\end{frame}

    \begin{column}{0.38\textwidth}

        \item<4-> Fer el canvi de variable \(a^x=t\).
        \item<7-> Resoldre i \textbf{desfer} el canvi de variable.
      \end{itemize}
    \end{column}

      \textbf{Exemple:}
      \footnotesize
      \setlength{\jot}{3pt}

    \end{column}

\end{frame}

\section{Funció logarítmica}

% ---------- DIAPO 4: Introducció + gràfic ----------
\begin{frame}[t]
  \frametitle{4. La funció logarítmica}

  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.57\textwidth}
      \begin{itemize}
        \item<1-> Com resolem expressions de la forma \(2^x=8\)?

        \item<3-> \(a^x=b \;\rightarrow\; x=\log_a b\)
      \end{itemize}
      \vspace{0.12cm}

        \end{caixadef}
      }
      \vspace{0.18cm}
      \only<7->{\raggedright Si \(f(x)=a^x\), llavors \(f^{-1}(x)=\log_a x\).\par}
      \only<8->{\raggedright Les gràfiques de \(f\) i \(f^{-1}\) són \textbf{simètriques} respecte \(y=x\).}
    \end{column}

  \small
  \begin{columns}[T,onlytextwidth]

        \end{itemize}
      }

      \vspace{0.15cm}
      \only<3->{

        \end{caixadef}
      }
      \only<4->{
        \begin{itemize}
          \item \(\log_a(3b)=\log_a 3+\log_a b\)
          \item \(\log_a 6=\log_a(2\cdot 3)=\log_a 2+\log_a 3\)
        \end{itemize}
      }
    \end{column}
    \begin{column}{0.48\textwidth}

      }

        \begin{itemize}
          \item \(\log_a\!\left(\dfrac{b}{3}\right)=\log_a b-\log_a 3\)
          \item \(\log_a 2=\log_a\!\left(\dfrac{6}{3}\right)=\log_a 6-\log_a 3\)
        \end{itemize}
      }

      \vspace{0.15cm}
      \only<7->{
        \begin{caixadef}[centrat]
          \(\log_a(x^p)=p\cdot\log_a x\)
        \end{caixadef}
      }
      \only<8->{
        \begin{itemize}
          \item \(\log_a(x^2)=2\log_a x\)
          \item \(2\log_a x=\log_a(x^2)\)
        \end{itemize}
      }
    \end{column}

% ---------- DIAPO 4.3: Canvi de base ----------
\begin{frame}[t]
  \frametitle{4.3. Canvi de base}

  \small
  \begin{itemize}
    \item<1-> Considerem \(b^x=m\) (equivalent a \(x=\log_b m\)).
    \item<2-> Si apliquem logaritmes en base \(a\) a cada costat:
    \only<2->{\[
      \log_a b^x=\log_a m \;\rightarrow\; x\log_a b=\log_a m \;\rightarrow\; x=\frac{\log_a m}{\log_a b}
    \]}
    \item<3-> \textbf{Fórmula del canvi de base:}
    \only<3->{\[
      \log_b m=\frac{\log_a m}{\log_a b}
    \]}
  \end{itemize}
\end{frame}

\begin{frame}[t]
  \frametitle{4.3. Canvi de base (exemples)}

  \small
  \begin{itemize}
    \item<1-> Moltes calculadores no tenen la tecla de logaritme en base arbitrària (\(\log_a\)). El canvi de base ens permet calcular \(\log_b m\) amb \(\ln\) o \(\log\):
    \only<1->{\[
      \log_3 5=\frac{\log 5}{\log 3}\approx\frac{0{,}6990}{0{,}4771}\approx 1{,}465
      \quad (3^{1{,}465}\approx 5)
    \]}
    \item<2-> També ens permet resoldre equacions:
    \only<2->{\[
      2^x=23 \;\rightarrow\; x=\log_2 23=\frac{\log 23}{\log 2}\approx 4{,}52
    \]}
  \end{itemize}
\end{frame}
